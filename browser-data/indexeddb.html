const db = new PouchDB('building-on-html5');

const add = async () => {
    const person = {
        _id: 'craig',
        twitter: 'craigshoemaker'
    };

    const response = await db.put(person);
    console.log(response);
};

const get = async () => {
    const craig = await db.get('craig');
    console.log(craig);
};

const update = async () => {
    const person = await db.get('craig');
    console.log(person);
    person.github = 'craigshoemaker';
    const response = await db.put(person);
    console.log(response);
};

const delete = async () => {
    const person = await db.get('craig');
    const response = await db.remove(person);
    console.log(response);
};
